@media (min-width: 768px) {
  .navbar-header {
    float: left;
  }
}
 
@media (min-width: 768px) {
  .navbar-toggle {
    display: none;
  }
}
 
@media (min-width: 768px) {
  .navbar-nav {
    float: left;
    margin: 0;
  }
   
  .navbar-nav > li {
    float: left;
  }

    padding-bottom: 15px;
  }
